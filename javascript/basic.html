# Symbol 类型

// 调用 Symbol 函数会返回一个 Symbol 的实例
// 不管调用多少次返回的实例都不相同
// 不可以用 new Symbol

// 全局 Symbol
let s1 = Symbol.for("foo");
let s2 = Symbol.for("foo");
// 此时 s1 和 s2 是同一个实例

// 反查全局 Symbol
Symbol.keyFor(s1); // foo
// 如果传递的不是一个 Symbol 那么会发生 TypeError
// 如果传递的不是全局 Symbol 那么会返回 undefined

// 可以使用 Symbol 作为 对象属性
let o = {
  [s1]: "foo val"
};
Object.defineProperty(o, s2, { value: "hello" });
Object.defineProperties(o, {
  [s3]: {},
  [s4]: {}
});

// Object.getOwnPropertyNames 返回常规属性
// Object.getOwnPropertySymbols 返回 Symbol 属性
// Object.getOwnPropertyDescriptors 返回两种
// Reflect.ownKeys 也会返回两种

/*
常用内置 Symbol
*/
Symbol.asyncIterator; // for await of
Symbol.hasInstance; // instanceof
Symbol.isConcatSpreadable; // concat 连接数组是否展开 可以用于 Set Object
Symbol.iterator; // for of
Symbol.match; // 字符串查找 正则
Symbol.replace; // 字符串替换 接受两个参数 字符串实例和要替换的字符串
Symbol.search; // 查找字符串位置
Symbol.species; // ?????

# 位操作符

// ES IEEE 754 64 位
// ES 在处理数值时会先把 64 位数值转换为 32 位数值
// 计算机中存放的数据是以补码形式存放的
// 对于有符号数来说第一位表示正负
// 正数的补码等于它本身
// 11111111
// 00000000
// 00000001

~1; // 对 1 按位非
// 0000 0001
// 1111 1110
// 1000 0001
// 1000 0010

240 & 15; // 按位与
// 1111 0000
// 0000 1111
// 0000 0000

240 | 15; // 按位或
// 1111 0000
// 0000 1111
// 1111 1111

12 ^ 8; // 按位异或
// 0000 1100
// 0000 1000
// 0000 0100    4

1 << 1; // 每次左移一位相当于 * 2， 可以用来去浮点
1 >>> 1; // 右移

  }

  console.log("努力给老板多买几套房");
  if ("星期天") {
    console.log("哎呀，今天要对不起老板了");
    continue;
  }

# with 语句
