# 冒泡排序

void bubble_sort(int *nums, int length)
{
    for (int i = 0; i < length; ++i)
    {
        int tmp;
        for (int j = 0; j < length - 1; ++j)
        {
            if (nums[j] > nums[j + 1])
            {
                tmp = nums[j];
                nums[j] = nums[j + 1];
                nums[j + 1] = tmp;
            }
        }
    }
}

# 选择排序

void select_sort(int *nums, int length)
{
    for (int i = 0; i < length; ++i)
    {
        int largest = i;
        int tmp;

        for (int j = i + 1; j < length; ++j)
        {
            if (nums[largest] < nums[j])
            {
                largest = j;
            }
        }

        if (largest != i)
        {
            tmp = nums[largest];
            nums[largest] = nums[i];
            nums[i] = tmp;
        }
    }
}

# 插入排序

void insert_sort(int *nums, int length)
{
    for (int i = 1; i < length; i++)
    {
        int j = i;
        int tmp = nums[i];
        while (tmp < nums[j - 1])
        {
            nums[j] = nums[j - 1];
            j--;
        }
        nums[j] = tmp;
    }
}

# 二分查找

void bin_search(int *nums, int length, int target)
{
    int l = 0;
    int r = length;
    int mid;
    while (l <= r)
    {
        mid = (l + r) / 2;
        printf("%d\n", mid);
        if (nums[mid] == target)
        {
            printf("index: %d - value: %d\n", mid, nums[mid]);
            break;
        }
        else if (nums[mid] < target)
        {
            l = mid + 1;
        }
        else
        {
            r = mid - 1;
        }
    }
}
